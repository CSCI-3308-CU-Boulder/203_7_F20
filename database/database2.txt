\\ creates user profile table
create table userprofile (user_id int primary key not null, firstname varchar not null, lastname varchar not null, email varchar not null
, password varchar not null, birthday varchar, signupdate date);

\\creates linking table to array of friends
create table friend_link (user_id int not null, friends_id_array int[], FOREIGN KEY(user_id) REFERENCES userprofile(user_id));

create table actions_link (user_id int not null, actions_id_array int[], FOREIGN KEY(user_id) REFERENCES userprofile(user_id));

create table achievments (user_id int not null, achievments_id_array int[], FOREIGN KEY(user_id) REFERENCES userprofile(user_id));

create table  achievments_list (achievments_id int primary key not null, achievment_name varchar not null);

create table actions_list (actions_id int not null, action_name varchar not null);
